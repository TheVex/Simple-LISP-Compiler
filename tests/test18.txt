(func Factorial (n)
  (cond (equal n 0)
    (return 1)
    (return (times n (Factorial (minus n 1))))))

(func Square (x)
  (times x x))

(setq x 5)
(setq y (Square x))
(setq z (Factorial 5))

(while (greater x 0)
  (setq x (minus x 1))
  (cond (equal x 2)
